(a) void reverseArray(int arr[], int n) {
    int start = 0, end = n - 1;
    while (start < end) {
        swap(arr[start], arr[end]);
        start++;
        end--;
    }
}

#include <iostream>
using namespace std;

(b) void multiplyMatrices(int A[10][10], int B[10][10], int C[10][10], int r1, int c1, int c2) {
    for(int i = 0; i < r1; i++) {
        for(int j = 0; j < c2; j++) {
            C[i][j] = 0;
            for(int k = 0; k < c1; k++) {
                C[i][j] += A[i][k] * B[k][j];
            }
        }
    }
}

(c)  void transposeMatrix(int A[10][10], int T[10][10], int rows, int cols) {
    for(int i = 0; i < rows; ++i)
        for(int j = 0; j < cols; ++j)
            T[j][i] = A[i][j];
}
